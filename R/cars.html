sp <- ggplot(cars, aes(x=Weight, y=MPG, size=Weight, color=Manufacturer)) + geom_point(alpha = 0.5)
sp + scale_color_manual(values=c("#999999", "#E69F00", "#56B4E9", "#C71585", "#4B0082"))
